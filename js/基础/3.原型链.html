原型链 #

一切皆为对象 #

  • 在javaScript中一切引用类型都可以看成为对象
  • 对象可以看成属性与方法的集合
  • 对象可以自由扩展属性
  • 对象都是通过函数（构造器）实例化出来的
javascript
const obj = {}
const arr = []
const fn = function () {}

obj.a = 1
arr.a = 1
fn.a = 1

console.log(obj.a) // 1
console.log(arr.a) // 1
console.log(fn.a) // 1

隐式原型 #

  • 每个对象都有一个隐式原型 __proto__ 属性，属性值是一个普通的对象
  • __proto__ 指向的该对象构造函数的prototype
  • prototype属性，是个对象,包含了{construct:指向他本身，__proto__:下个原型}
  • Object.prototype__proto__指向的是null
javascript
obj.__proto__ ===Object.prototype// 一个对象的__proto__指向他的构造器的原型
obj.__proto__.constructor==obj.constructor ===Object.prototype.constructor===Object//原型对象的构造器指向自己
javascript
/**
 * 1.每个对象都有一个__proto__属性
 * 2.每个类都是一个prototype属性，是个对象,包含了1.construct指向他本身，2.__proto__下个原型
 * 3.一个对象的__proto__属性指向的是该对象所属类的prototype属性
 * 4.对象的属性查找，会先从该对象的实例实行开始，让后沿着__proto__链查找
 * 5.Function 与 Object，Array,String 既是对象又是函数，Object的__proto__指向的是Function的prototype,而Function的__proto__指向的是自己Function的prototype
 * 6.Object.__proto__ === null,原型链的终点是Object.__proto__ ->null
 * 7.hasOwnProperty查找的是实例属性，in 查找的是原型链属性
 * 例子
 *  {} -> Object.prototype -> null
 *  animal -> Animal.prototype -> Object.prototype -> null
 *  [] -> Array.prototype ->Object.prototype -> null
 *  Array -> Function.prototype ->  Object.prototype -> null
 *  Function.__proto__=== Function.prototype
 *  Object.__proto__ === Function.prototype
 * */

function Animal() {
  this.type = 1;
}
Animal.prototype.type = 2;
Object.prototype.type = 3;
let animal = new Animal();
let obj = {
  age: 123
};
Function.prototype.name = 11;
Function.prototype.aa = 22;
console.log(Animal.name);
console.log(Animal.aa);
let arr = new Array();
console.log(Animal.__proto__ === Function.prototype);
console.log(Array.__proto__ === Function.prototype);
console.log([].__proto__.__proto__ === Object.prototype);

console.log(Object.__proto__ === Function.prototype);
console.log(Function.__proto__ === Function.prototype);
console.log(Function.prototype.__proto__ === Object.prototype);
console.log(Object.prototype.__proto__ === null);// 原型链的终点是

console.log(Object.__proto__ === Function.prototype); // Object instanceof Function
console.log(Function.__proto__ === Function.prototype); //  Function instanceof Function
console.log(Function.__proto__.__proto__ === Object.prototype); // Function instanceof Object
console.log(Object.__proto__.__proto__ === Object.prototype); // Object instanceof Object
console.log({}.__proto__.__proto__ === Object.prototype);

typeof数据类型检测 #

  • 检测数据类型 typeof返回的都是字符串

  • 基本数据类型 number string boolean undefined symbol

  • 引用类型 null {} [] /&$/ Date => object

  • typeof function => function

javascript
console.log(typeof a);    // undefined
console.log(typeof 1);   // number
console.log(typeof 'zhufeng'); // string
console.log(typeof true);  // boolean
console.log(typeof Symbol('a'));  // symbol

console.log(typeof function () { });  //function

console.log(typeof [1, 2, 3]);  //object
console.log(typeof { name: 'zhufeng' });  //object
console.log(typeof null);  //object
console.log(typeof new Number(1));  //object

instanceof 判断对象是不是一个构造器的实例 #

  • instanceof运算符的第一个参数是一个对象，第二个参数一般是一个函数
  • instanceof的判断规则是: 沿着对象的__proto__这条链来向上查找找,如果能找到函数的prototype则返回true,否则 返回false
java
function _instanceof(a, b) {
  let leftProto = a.__proto__;// 对象
  let rightPrototype = b.prototype;// 构造器
  while (leftProto) {
    if (leftProto === rightPrototype) {
      return true;
    }
    leftProto = leftProto.__proto__;
  }
  return false;
}
// test
console.warn(_instanceof([], Array));
console.warn(_instanceof([], Object));
function Animal() {
  this.a = 1;
}
let animal = new Animal();
console.warn(_instanceof(animal, Animal));

function Father(name) {
  this.name = name;
}
Father.staticFatherName = "FatherName";
Father.staticGetFatherName = function () {
  console.log(Father.staticFatherName);
  return Father.staticFatherName;
};
Father.prototype.aa = "aa";
function Son(name, age) {
  Father.call(this, name); // 调用父类构造函数
  Object.setPrototypeOf(Son, Father); // 继承静态属性
  this.age = age;
}
Son.prototype = Object.create(Father.prototype); // 设置原型链,继承父类属性
Son.prototype.constructor = Son; // 设置构造器
Son.staticSonName = "son name";
Son.staticGetSonName = function () {
  console.log(Son.staticSonName);
  return Son.staticSonName;
};
let son = new Son("zs", 10);
console.log(son.age, son.name, son.aa); // 10 zs aa
console.log(Son.staticSonName, Son.staticFatherName); // son name FatherName
console.log(son.constructor === Son); // true
console.log(son instanceof Son, son instanceof Father, son instanceof Object); // true true true
