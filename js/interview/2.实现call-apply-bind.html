实现call-apply-bind #

call、apply、bind是用来改变this指向的

请参考上下文中this指向

差异：

  • call,apply的第一个参数都是thisArgs,要绑定的this对象，之后参数有不同，call是通过参数列表传递，apply通过数组传递参数
  • bind返回的是一个已经改变this的新函数

实现call #

javascript
Function.prototype.call = function() {
  // 类型判断，只有函数才能调用
  if (typeof this !== "function") {
    throw Error("call method must be used by function");
  }
  // 参数获取  
  const args = Array.from(arguments);
  const context = args[0] || window; // 上下文 this指向
  const fn = Symbol();
  context[fn] = this;
  // 通过对象的点调用函数改变this指向  
  const res = context[fn](...args.slice(1));
  delete context[fn];// 上传属性
  return res;
};

function foo() {
  console.log(this);
  console.log(arguments);
}
let obj = {
  name: "zs",
  age: 12
};

foo.call(obj,1,2)

实现apply #

javascript
Function.prototype.apply = function() {
  // 类型判断，只有函数才能调用
  if (typeof this !== "function") {
    throw Error("call method must be used by function");
  }
  // 参数获取  
  const args = Array.from(arguments);
  const context = args[0] || window; // 上下文 this指向
  const fn = Symbol();
  context[fn] = this;
  // 通过对象的点调用函数改变this指向  
  const res = context[fn](...args[1];
  delete context[fn];// 上传属性
  return res;
};

function foo() {
  console.log(this);
  console.log(arguments);
}
let obj = {
  name: "zs",
  age: 12
};

foo.apply(obj,[1,2])

实现bind #

java
Function.prototype.bind = function() {
	const args = Array.from(arguments);
	const context = args[0];
	const _this = this;
	return function fn() {
		if (this instanceof fn) {
			return new _this(...args.slice(1).concat(...arguments));
		} else {
			return _this.apply(context, args.slice(1).concat(...arguments));
		}
	};
};

let obj = {
	name: '若川',
};
function original(a, b){
	console.log(this.name);
	console.log([a, b]);
	return false;
}
let bound = original.bind(obj, 1);
console.log(bound)
let boundResult = bound(2); // '若川', [1, 2]
console.log(boundResult)
